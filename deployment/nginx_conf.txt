upstream app_server {
    server unix:/home/ambigo_in/server/deployment/run/gunicorn.sock fail_timeout=0;
}

server {
    listen 443 ssl;
    server_name ambigo.in;

    # SSL configuration (ensure you have valid certificates)
    ssl_certificate /home/ambigo_in/server/certificates/cert.pem;
    ssl_certificate_key /home/ambigo_in/server/certificates/key.pem;
    
    # Optional: Add SSL protocols and ciphers for more security
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers 'TLS_AES_128_GCM_SHA256:TLS_AES_256_GCM_SHA384:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-RSA-AES256-GCM-SHA384';

    client_max_body_size 20M;

    # Access and Error Logs
    access_log /home/ambigo_in/server/deployment/logs/nginx-access.log;
    error_log /home/ambigo_in/server/deployment/logs/nginx-error.log;

    # Limit HTTP Methods to GET and POST only
    location / {
        limit_except GET POST {
            deny all;
        }

        # Proxy requests to the upstream server (Gunicorn)
        proxy_pass http://app_server;
	    client_max_body_size 20M;

        # Headers to process websockets
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        # Optional: Add headers to improve security
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location /docs {
        allow 127.0.0.1;            # Allow localhost
        allow 192.168.1.100;        # Allow IP 192.168.1.100
        deny all;                   # Deny access to everyone else
           
        # Proxy requests to the upstream server (Gunicorn)
        proxy_pass http://app_server;
    }

    location /redoc {
        allow 127.0.0.1;            # Allow localhost
        allow 192.168.1.100;        # Allow IP 192.168.1.100
        deny all;                   # Deny access to everyone else

        # Proxy requests to the upstream server (Gunicorn)
        proxy_pass http://app_server;
    }

    location /openapi.json {
        allow 127.0.0.1;            # Allow localhost
        allow 192.168.1.100;        # Allow IP 192.168.1.100
        deny all;                   # Deny access to everyone else

        # Proxy requests to the upstream server (Gunicorn)
        proxy_pass http://app_server;
    }
}
